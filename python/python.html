# python 3

cnt = 0

for i in range(1, 5):
	for j in range(1, 5):
		for k in range(1, 5):
			if i != j and j != k and j != i:
				print(i * 100 + j * 10 + k)
				cnt += 1
print('count:', cnt)

# python

s = 8950
arr = [1000000, 600000, 400000, 200000, 100000, 0]
rat = [0.01, 0.015, 0.03, 0.05, 0.075, 0.1]
e = 0

for i in range(0, 6):
	if s > arr[i]:
		e += (s - arr[i]) * rat[i]
		print((s - arr[i]) * rat[i])
		s = arr[i]
